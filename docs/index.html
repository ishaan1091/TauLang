Example Code

See TauLang in action with real examples

Basic Operations 
// Variables and arithmetic
sun_liyo_tau x ne_bana_diye 10;
sun_liyo_tau y ne_bana_diye 20;
sun_liyo_tau sum ne_bana_diye x + y;
print(sum); // prints 30

// Conditionals
agar_maan_lo (sum > 25) {
    print("Greater than 25");
} na_toh {
    print("Less than or equal to 25");
}

// While loops
sun_liyo_tau i ne_bana_diye 0;
jab_tak (i < 5) {
    i ne_bana_diye i + 1;
    print(i);
}
Functions and Closures 
// Function definition
sun_liyo_tau add ne_bana_diye tau_ka_jugaad(a, b) {
    laadle_ye_le a + b;
};

print(add(5, 3));  // prints 8

// Higher-order functions
sun_liyo_tau apply ne_bana_diye tau_ka_jugaad(fn, x) {
    laadle_ye_le fn(x);
};

// Closures
sun_liyo_tau makeCounter ne_bana_diye tau_ka_jugaad() {
    sun_liyo_tau count ne_bana_diye 0;
    laadle_ye_le tau_ka_jugaad() {
        count ne_bana_diye count + 1;
        laadle_ye_le count;
    };
};

sun_liyo_tau counter ne_bana_diye makeCounter();
print(counter());  // prints 1
print(counter());  // prints 2
Arrays and Hash Maps 
// Arrays
sun_liyo_tau arr ne_bana_diye [1, 2, 3, 4, 5];
print(arr[0]);              // prints 1
arr[2] ne_bana_diye 10;     // Modify element
print(len(arr));            // prints 5
print(first(arr));          // prints 1
print(last(arr));           // prints 5

// Hash maps
sun_liyo_tau person ne_bana_diye {
    "name": "Tau",
    "age": 50,
    "city": "Mumbai"
};

print(person["name"]);                    // prints "Tau"
person["country"] ne_bana_diye "India";   // Add new key
person["age"] ne_bana_diye 51;            // Update key
Recursive Functions 
// Factorial
sun_liyo_tau factorial ne_bana_diye tau_ka_jugaad(n) {
    agar_maan_lo (n <= 1) {
        laadle_ye_le 1;
    } na_toh {
        laadle_ye_le n * factorial(n - 1);
    }
};

print(factorial(5));  // prints 120

// Fibonacci
sun_liyo_tau fibonacci ne_bana_diye tau_ka_jugaad(n) {
    agar_maan_lo (n <= 1) {
        laadle_ye_le n;
    } na_toh {
        laadle_ye_le fibonacci(n - 1) + fibonacci(n - 2);
    }
};

print(fibonacci(10));  // prints 55

Get Started

Install and start coding in TauLang today

1

Clone the Repository

git clone https://github.com/ishaan1091/taulang.git
cd taulang
2

Build the Interpreter

make build
3

Start Coding

# Run REPL
./bin/taulang

# Or execute a file
./bin/taulang program.tau

Requirements

  • Go 1.21 or higher
  • Make (optional, for Makefile commands)
